146:     def parse_arguments(arguments)
147:       @resource_path, @opts, skip_next = [], {}, false
148:       @opts.update(options[:query]) if options[:query]
149: 
150:       arguments.each_with_index { |arg, index|
151:         if skip_next

154:         end
155: 
156:         if arg =~ OPTION_RE
157:           key, value, next_arg = $1, $2, arguments[index + 1]

160:           else
161:             skip_next = true
162:             next_arg
163:           end
164:         else
165:           resource_path << arg

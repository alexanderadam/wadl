Class WADL::Address
In: lib/wadl.rb
Parent: Object

The Address class keeps track of the user‘s path through a resource graph. Values for WADL parameters may be specified at any time using the bind method. An Address cannot be turned into a URI and header set until all required parameters have been bound to values.

An Address object is built up through calls to Resource#address

448:       query_var_values  = args[:query]   || {}
449:       header_var_values = args[:headers] || {}

451:       # Bind variables found in the path fragments.
452:       path_params_to_delete = []
453: 
454:       path_fragments.each { |fragment|
455:         if fragment.respond_to?(:to_str)
456:           # This fragment is a string which might contain {} substitutions.
457:           # Make any substitutions available to the provided path variables.
458:           self.class.embedded_param_names(fragment).each { |param_name|
459:             value = path_var_values[param_name] || path_var_values[param_name.to_sym]
460: 
461:             value = if param = path_params[param_name]
462:               path_params_to_delete << param
463:               param % value
464:             else
465:               Param.default.format(value, param_name)
466:             end
467: 
468:             fragment.gsub!("{#{param_name}}", value)
469:           }
470:         else
471:           # This fragment is an array of Param objects (style 'matrix'
472:           # or 'plain') which may be bound to strings. As substitutions
473:           # happen, the array will become a mixed array of Param objects
474:           # and strings.
475:           fragment.each_with_index { |param, i|
476:             if param.respond_to?(:name)
477:               name = param.name
478: 
479:               value = path_var_values[name] || path_var_values[name.to_sym]
480:               value = param % value
481:               fragment[i] = value if value
482: 
483:               path_params_to_delete << param
484:             end
485:           }
486:         end
487:       }
488: 
489:       # Delete any embedded path parameters that are now bound from
490:       # our list of unbound parameters.
491:       path_params_to_delete.each { |p| path_params.delete(p.name) }
492: 
493:       # Bind query variable values to query parameters
494:       query_var_values.each { |name, value|
495:         param = query_params.delete(name.to_s)
496:         query_vars << param % value if param
497:       }
498:

500:       header_var_values.each { |name, value|
501:         param = header_params.delete(name.to_s)

503:       }

184:         attributes = element.attributes
185: 
186:         me = new
187:         me.parent = parent
188: 
189:         @collections.each { |name, klass|
190:           me.instance_variable_set("@#{klass.names[:collection]}", [])
191:         }
192:

194:           # Handle objects that are just references to other objects

197:           href.sub!(/\A#/, '') or warn "Warning: HREF #{href} should be ##{href}"
198: 
199:           me.attributes['href'] = href
200:         else
201:           # Handle this element's attributes
202:           @required_attributes.each { |name|
203:             name = name.to_s
204: 
205:             raise ArgumentError, %Q{Missing required attribute "#{name}" in element: #{element}} unless attributes[name]
206: 
207:             me.attributes[name] = attributes[name]
208:             me.index_key = attributes[name] if name == @index_attribute
209:           }
210: 
211:           @attributes.each { |name|
212:             name = name.to_s
213: 
214:             me.attributes[name] = attributes[name]
215:             me.index_key = attributes[name] if name == @index_attribute
216:           }
217:         end
218: 
219:         # Handle this element's children.
220:         if @contents_are_mixed_data
221:           me.instance_variable_set(:@contents, element.children)
222:         else

224:             if klass = @members[child.name] || @collections[child.name]
225:               object = klass.from_element(me, child, need_finalization)
226: 
227:               if klass == @members[child.name]
228:                 instance_variable_name = "@#{klass.names[:member]}"

231:                   raise "#{name} can only have one #{klass.name}, but several were specified in element: #{element}"
232:                 end
233: 
234:                 me.instance_variable_set(instance_variable_name, object)
235:               else
236:                 me.instance_variable_get("@#{klass.names[:collection]}") << object
237:               end
238:             end
239:           }
240:         end

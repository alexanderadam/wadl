# File lib/wadl/has_docs.rb, line 39
def define_singleton(r, sym, method)
  name = r.send(sym)

  if name && name !~ /\W/ && !r.respond_to?(name) && !respond_to?(name)
    instance_eval(%{def #{name}\n#{method}('#{name}')\nend})
  end
end
